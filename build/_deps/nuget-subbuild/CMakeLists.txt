                    SOURCE_DIR          "C:/Users/pc/Desktop/Mon_Vocabulaire_Flutter/build/_deps/nuget-src"
                    BINARY_DIR          "C:/Users/pc/Desktop/Mon_Vocabulaire_Flutter/build/_deps/nuget-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)

             "<DOWNLOADED_FILE>" "C:/Users/pc/Desktop/Mon_Vocabulaire_Flutter/build/_deps/nuget-src"

  DEPENDERS  configure
  BYPRODUCTS "C:/Users/pc/Desktop/Mon_Vocabulaire_Flutter/build/_deps/nuget-src/${dlFileName}"
  COMMENT    "Copying file to SOURCE_DIR"
)
